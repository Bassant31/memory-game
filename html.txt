        <script>
        
        function picRotate(id){
         document.getElementById(id).style.transform="rotateY(180deg)"
         ;
            
            
        
    
            
         
        }
               var flag=false;
               
                var score=0;
                var fail=0;
             var firstCard=null;
                var secondCard=null;
            var firstID=null;
            var secondID=null;
            
            function calcScore(id,name){
                
                if(fail<3){
                picRotate(id);
                
             
                
                if (flag==false){
                    firstCard=name;
                    firstID=id;
                    flag=true;
                    
                }
                else{
                     secondCard=name;
                    secondID=id;
                    flag=false;
                    
                }
                if((firstCard==secondCard) && (secondCard!=null)&&(firstID!=secondID)){
                    score++;
                    
                 firstCard=null;
                secondCard=null;
                   document.getElementById("score").innerHTML=score;
                    document.getElementById("fail").innerHTML=fail;
                      
                    firstID=null;
                    secondID=null;
                    
                    
                }
            
                if((firstCard!=secondCard) && (secondCard!=null)&&(firstID!=secondID)){
                    
                    fail++;
                    
                    
                 firstCard=null;
                secondCard=null;
                    
                if(fail==3){
                        document.getElementById("gameOver").innerHTML="Game Over";

                    }
                    
                    
                      setTimeout(function(){document.getElementById(firstID).style.transform="rotateY(180deg)"; }, 3000);
                    
                     setTimeout(function(){document.getElementById(secondID).style.transform="rotateY(180deg)";} , 3000);
                    
                     
                        
                    
                    
                    firstID=null;
                    secondID=null;
                    
                    document.getElementById("score").innerHTML=score;
                    document.getElementById("fail").innerHTML=fail;
                    
                }
                 
            }
                   
               
                
                
                
               
            }
